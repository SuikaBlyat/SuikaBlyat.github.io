#import所需的各种库
from selenium import webdriver
from selenium.webdriver.chrome.options import Options
import bs4
import time
from openpyxl import Workbook
import requests


article_url_list = [] 
article_title_list = []  #这个好像多余了，但我懒得删掉了
pages = int(input('请输入需要爬取文章页数：'))
author = input('请输入专栏作者：')  #注意这里不是昵称，是作者的英文id哦
headers = {
            'origin': 'https://zhuanlan.zhihu.com',
            'referer': 'https://zhuanlan.zhihu.com/{}'.format(author),
            'user-agent': 'Mozilla/5.0 (Windows NT 10.0; WOW64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/78.0.3904.108 Safari/537.36'
}


#这部分没封装启动函数是因为封装了之后crawler函数会发现driver undefined，我图省事就取消封装了，没有尝试__init__,反正能用就是hhh
wb = Workbook() #创建excel工作簿
ws = wb.active  #定义活动工作表
ws.append(['标题', '简介', '链接', '赞同'])
print('_______\n正在启动浏览器')
chrome_options = Options()
chrome_options.add_argument('--headless')
driver = webdriver.Chrome(options=chrome_options)
driver.get('https://zhuanlan.zhihu.com/{}'.format(author))  # 专栏地址
time.sleep(2) #给点时间加载


#封装滑动函数
def scroll():
    print('_______\n正在模拟Chrome下滑')
    i = 0
    while i < pages:  # 当文章数量多时，
        driver.execute_script("window.scrollBy(0,5000)") #滑动幅度可自行修改，5000还挺大的
        time.sleep(1)
        i += 1
    print('_______\n模拟结束')


def crawler_list():
    print('_______\n正在爬取列表:\n')
    res = driver.page_source
    bs_res = bs4.BeautifulSoup(res, 'html.parser')
    articles = bs_res.find_all('li', class_='ArticleItem')
    for article in articles:
        title = article.find_all('h3', class_='ArticleItem-Title')[0].text
        article_url = article.find_all('a')[0]['href']
        summary = article.find('p', class_='RichContent-inner').text
        vote = article.find('button', class_='Button VoteButton VoteButton--up')['aria-label']
        vote = vote[2:]
        article_url_list.append(article_url) #添加到URL列表方便下一个函数爬取文章内容
        article_title_list.append(title) #回应开头那个多余的代码）
        line = [title, summary, article_url, vote]
        ws.append(line)
        print('添加《{title}》到列表'.format(title=title))
        time.sleep(1) #可删，不过就看不到一个一个title出来了
    print('_______\n列表爬取完毕\n')


def crawler_article():
    print('_______\n正在录入内容:\n')
    for z in range(len(article_url_list)):
        url_2= article_url_list[z]
        r = requests.get(url_2, headers=headers)
        bs_r = bs4.BeautifulSoup(r.text, 'html.parser')
        content = bs_r.find(class_='RichText ztext Post-RichText') #需要纯文本的请在后面加.text 然后可以把下一行删掉
        content = str(content)
        title = bs_r.find_all('h1', class_='Post-Title')[0].text
        title_short = title[:7] #截取纯数字
        f = open('SourceCodes/{}.txt'.format(title_short), 'w', encoding='utf-8')
        f.write(content)
        f.close()
        print('已录入《{}》'.format(title))


def close():
    wb.save('SumList.xlsx')
    print('_______\n已保存文章列表') 
    wb.close()
    driver.close()
    print('_______\n驱动已关闭\nExcel已关闭\n_______')
    input('\n请输入回车结束爬虫') #不在pycharm运行时像暂时不结束脚本才加的，可删


def main():
    scroll()
    crawler_list()
    crawler_article()
    close()


main()
